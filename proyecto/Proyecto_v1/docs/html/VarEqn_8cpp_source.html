35 Matrix v(3, 1);
36 r(1) = yPhi(1); r(2) = yPhi(2); r(3) = yPhi(3);

42 for (int j = 1; j <= 6; j++) {

44 Phi(i, j) = yPhi(6*j + i);
45 }
46 }

48 // Acceleration and gradient
49 Matrix a = AccelHarmonic(r, E, auxParam.n, auxParam.m);
50 Matrix G = G_AccelHarmonic(r, E, auxParam.n, auxParam.m);
51
52 // Time derivative of state transition matrix
53 Matrix yPhip(42, 1);
54 Matrix dfdy(6, 6);
55
56 for (int i = 1; i <= 3; i++) {
57 for (int j = 1; j <= 3; j++) {
58 dfdy(i, j) = 0.0; // dv/dr(i,j)
59 dfdy(i+3, j) = G(i, j); // da/dr(i,j)
60 if (i == j) {
61 dfdy(i, j+3) = 1.0;
62 } else {
63 dfdy(i, j+3) = 0.0; // dv/dv(i,j)
64 }
65 dfdy(i+3, j+3) = 0.0; // da/dv(i,j)
66 }
67 }
68
69 Matrix Phip = dfdy * Phi;
70
71 // Derivative of combined state vector and state transition matrix
72 for (int i = 1; i <= 3; i++) {
73 yPhip(i) = v(i); // dr/dt(i)
74 yPhip(i+3) = a(i); // dv/dt(i)
75 }
76
77 for (int i = 1; i <= 6; i++) {
78 for (int j = 1; j <= 6; j++) {
79 yPhip(6*j + i) = Phip(i, j); // dPhi/dt(i,j)
80 }
81 }
82
83 return yPhip;
