7void elements(const Matrix& y, double& p, double& a, double& e, double& i,
8 double& Omega, double& omega, double& M)
9{
10 const double pi2 = 2.0 * M_PI;
11
12 Matrix r(3, 1);
13 Matrix v(3, 1);
14 r(1) = y(1); r(2) = y(2); r(3) = y(3); // Position
15 v(1) = y(4); v(2) = y(5); v(3) = y(6); // Velocity
16
17 // Areal velocity
18 Matrix h = Matrix::cross(r, v);
19 double magh = h.norm();
20 p = magh * magh / GM_Earth;
21 double H = h.norm();
22
23 // Long. ascend. node
24 Omega = std::atan2(h(1), -h(2));
25 Omega = std::fmod(Omega, pi2);
26 if (Omega < 0) Omega += pi2; // me aseguro que pos
27
28 // Inclination
29 i = std::atan2(std::sqrt(h(1)*h(1) + h(2)*h(2)), h(3));
30
31 // Arg. of latitude
32 double u = std::atan2(r(3)*H, -r(1)*h(2) + r(2)*h(1));
33
34 // Distance
35 double R = r.norm();
36
37 // Semi-major axis
38 double v_squared = Matrix::dot(v, v);
39 a = 1.0 / (2.0/R - v_squared/GM_Earth);
40
41 // e*cos(E) y e*sin(E)
42 double eCosE = 1.0 - R/a;
43 double eSinE = Matrix::dot(r, v) / std::sqrt(GM_Earth * a);
44
45 // Eccentricity
46 double e2 = eCosE*eCosE + eSinE*eSinE;
47 e = std::sqrt(e2);
48
49 // Eccentric anomaly
50 double E = std::atan2(eSinE, eCosE);
51
52 // Mean anomaly
53 M = std::fmod(E - eSinE, pi2);
54 if (M < 0) M += pi2; // Ensure positive
55
56 // True anomaly
57 double nu = std::atan2(std::sqrt(1.0 - e2) * eSinE, eCosE - e2);
58
59 // Arg. of pericenter
60 omega = std::fmod(u - nu, pi2);
61 if (omega < 0) omega += pi2; // pos
62}
