38 UT1_UTC = eop(7, i) + (eop(7, i+1) - eop(7, i)) * fixf;

41 deps = eop(10, i) + (eop(10, i+1) - eop(10, i)) * fixf;
42 dx_pole = eop(11, i) + (eop(11, i+1) - eop(11, i)) * fixf;
43 dy_pole = eop(12, i) + (eop(12, i+1) - eop(12, i)) * fixf;
44 TAI_UTC = eop(13, i);
45
46 // arcsegundos -> radianes
47 x_pole = x_pole / Arcs;
48 y_pole = y_pole / Arcs;
49 dpsi = dpsi / Arcs;
50 deps = deps / Arcs;
51 dx_pole = dx_pole / Arcs;
52 dy_pole = dy_pole / Arcs;
53
54 } else { // interp == 'n' (sin interpolacion)
55
56 x_pole = eop(5, i) / Arcs;
57 y_pole = eop(6, i) / Arcs;
58 UT1_UTC = eop(7, i);
59 LOD = eop(8, i);
60 dpsi = eop(9, i) / Arcs;
