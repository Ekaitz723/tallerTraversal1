126 double l = fmod(485866.733 + (1325.0*rev + 715922.633)*T + 31.310*T2 + 0.064*T3, rev);
127 double lp = fmod(1287099.804 + (99.0*rev + 1292581.224)*T - 0.577*T2 - 0.012*T3, rev);
128 double F = fmod(335778.877 + (1342.0*rev + 295263.137)*T - 13.257*T2 + 0.011*T3, rev);
129 double D = fmod(1072261.307 + (1236.0*rev + 1105601.328)*T - 6.891*T2 + 0.019*T3, rev);
130 double Om = fmod(450160.280 - (5.0*rev + 482890.539)*T + 7.455*T2 + 0.008*T3, rev);
131
132 // Nutation in longitude and obliquity [rad]
133 dpsi = 0.0;
134 deps = 0.0;
135
136 for (int i = 0; i < N_coeff; i++) {
137 double arg = (C[i][0]*l + C[i][1]*lp + C[i][2]*F + C[i][3]*D + C[i][4]*Om) / Arcs;
138 dpsi += (C[i][5] + C[i][6]*T) * sin(arg);
139 deps += (C[i][7] + C[i][8]*T) * cos(arg);
140 }
141
142 dpsi = 1.0e-5 * dpsi / Arcs;
143 deps = 1.0e-5 * deps / Arcs;

146void test_NutAngles()

150 // Test 1: J2000.0 epoch (MJD 51544.5) - T=0
151 NutAngles(51544.5, dpsi, deps);

153 // En J2000.0, la nutacion debe ser pequeña pero no cero
154 assert(fabs(dpsi) < 1e-4); // Menos de 1e-4 radianes (~20 arcseg)
155 assert(fabs(deps) < 1e-4); // Similar para oblicuidad
156
157 // Test 2: Verificar que los valores estan en rangos tipicos de nutacion
158 // La nutacion tipicamente varia entre ±1 arcsegundo (~±5e-6 rad)
159 assert(fabs(dpsi) < 1e-3); // Rango conservador
160 assert(fabs(deps) < 1e-3);
161
162 // Test 3: Epoca diferente - 1 año despues de J2000
163 NutAngles(51910.0, dpsi, deps); // ~1 año = 365.25 dias
164
165 assert(fabs(dpsi) < 1e-3);
166 assert(fabs(deps) < 1e-3);
167
168 // Test 4: Verificar que los valores cambian con el tiempo
169 double dpsi1, deps1, dpsi2, deps2;
170 NutAngles(51544.5, dpsi1, deps1); // J2000.0
171 NutAngles(52000.0, dpsi2, deps2); // ~1.25 años despues

173 // Los valores deben ser diferentes debido al movimiento lunar/solar
174 assert(fabs(dpsi1 - dpsi2) > 1e-8);
175 assert(fabs(deps1 - deps2) > 1e-8);
176
177 // Test 5: Epoca historica - 1980 (MJD ~44240)
178 NutAngles(44239.0, dpsi, deps); // 1 enero 1980
179
180 assert(fabs(dpsi) < 1e-3);
181 assert(fabs(deps) < 1e-3);
182
183 // Test 6: Epoca futura - 2050 (MJD ~69807)
184 NutAngles(69807.0, dpsi, deps); // ~1 enero 2050
185
186 assert(fabs(dpsi) < 1e-3);
187 assert(fabs(deps) < 1e-3);
