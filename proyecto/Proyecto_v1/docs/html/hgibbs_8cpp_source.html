9void hgibbs(const Matrix& r1, const Matrix& r2, const Matrix& r3,
10 double Mjd1, double Mjd2, double Mjd3,
11 Matrix*& v2, double& theta, double& theta1, double& copa, std::string& error)
12{
13 error = " ok";
14 theta = 0.0;
15 theta1 = 0.0;
16
17 double magr1 = r1.norm();
18 double magr2 = r2.norm();
19 double magr3 = r3.norm();
20
21 v2 = new Matrix(3, 1);
22
23 double tolangle = 0.01745329251994;
24 double dt21 = (Mjd2 - Mjd1) * 86400.0;
25 double dt31 = (Mjd3 - Mjd1) * 86400.0;
26 double dt32 = (Mjd3 - Mjd2) * 86400.0;
27
28 Matrix p = Matrix::cross(r2, r3);
29 Matrix pn = unit(p);
30 Matrix r1n = unit(r1);
31 copa = std::asin(Matrix::dot(pn, r1n));
32
33 double dot_check = Matrix::dot(r1n, pn);
34 // printf("DEBUG: abs(dot(r1n,pn)) = %.6f, limit = 0.017452406\n", std::abs(dot_check));
35
36 if (std::abs(Matrix::dot(r1n, pn)) > 0.017452406) {
37 // printf("DEBUG: Setting error to 'not coplanar'\n");
38 error = "not coplanar";
39 }
40
41 theta = angl(r1, r2);
42 theta1 = angl(r2, r3);
43
44 if ((theta > tolangle) || (theta1 > tolangle)) {
45 error = " angl > 1ø";
46 }
47
48 double term1 = -dt32 * (1.0 / (dt21 * dt31) + GM_Earth / (12.0 * magr1 * magr1 * magr1));
49 double term2 = (dt32 - dt21) * (1.0 / (dt21 * dt32) + GM_Earth / (12.0 * magr2 * magr2 * magr2));
50 double term3 = dt21 * (1.0 / (dt32 * dt31) + GM_Earth / (12.0 * magr3 * magr3 * magr3));
51
52 *v2 = r1 * term1 + r2 * term2 + r3 * term3;
53}
